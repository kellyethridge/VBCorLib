@examples The example demonstrates instantiating a FileStream object using two methods. 
Public Sub Main()
    Dim FS As FileStream
    
    ' open file using global method.
    Set FS = NewFileStream("MyFile.txt", FileMode.OpenExisting)
    
    ' open file using global Cor accessor.
    Set FS = Cor.NewFileStream("MyFile.txt", FileMode.OpenExisting)
End Sub
