@examples 
Public Sub Main()
    Dim List As New ArrayList
    
    List.Add "Hello"
    List.Add "World"
    List.Add "!"
    
    Debug.Print "List"
    Debug.Print CorString.Format("    Count:    {0}", List.Count)
    Debug.Print CorString.Format("    Capacity: {0}", List.Capacity)
    Debug.Print "      Values:"
    PrintValues List
End Sub

Private Sub PrintValues(ByVal List As IEnumerable)
    Dim Value As Variant
    
    For Each Value In List
        Debug.Print "  " & Value;
    Next
End Sub

' This code produces the following:
'
' List
'     Count:    3
'     Capacity: 16
'     Values:  Hello  World  !
