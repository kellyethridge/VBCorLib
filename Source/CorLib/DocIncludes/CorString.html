@examples The following example shows how to call various methods on the CorString class. 
Public Sub Main()
    Const Sentance As String = "This string consists of a single short sentence."
    
    Dim Words() As String
    Dim Word As Variant
    
    Words = CorString.Split(Sentance, " ")
    
    For Each Word In Words
        Debug.Print CorString.Format("'{0}'", Word)
    Next
End Sub

' This code produces the following output.
'
'    'This'
'    'string'
'    'consists'
'    'of'
'    'a'
'    'single'
'    'short'
'    'sentence.'
