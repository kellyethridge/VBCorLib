@examples The following example demonstrates using the AddRange method to add a collection of items. 
Private Sub Main()
    Dim List As New ArrayList
    Dim Words() As String
    
    Words = NewStrings("The", "fox", "jumps", "over", "the", "dog")
    
    List.AddRange Words
    
    PrintValues List
End Sub

Private Sub PrintValues(ByVal Source As IEnumerable)
    Dim Value As Variant
    
    Console.WriteValue vbTab
    
    For Each Value In Source
        Debug.Print CorString.Format("{0} ", Value);
    Next Value
End Sub

' This example produces the following output.
'
'    The fox jumps over the dog
